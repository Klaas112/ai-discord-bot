# ID van het log kanaal
LOG_CHANNEL_ID = 123456789012345678  # 1470421288479887380

intents = discord.Intents.default()
intents.message_content = True
intents.members = True  # nodig voor ban

client = discord.Client(intents=intents)

@client.event
async def on_ready():
    print(f'Bot online als {client.user}')

@client.event
async def on_message(message):
    # bot reageert niet op zichzelf
    if message.author == client.user:
        return

    # bot reageert alleen als hij gepingd wordt
    if client.user in message.mentions:

        content = message.content.lower()
        log_channel = client.get_channel(LOG_CHANNEL_ID)

        # BAN command
        if "ban" in content and "reason" in content:
            try:
                parts = content.split("ban")[1].split("reason")
                member_name = parts[0].strip().split()[0]  # naam van gebruiker
                reason = parts[1].strip()

                # zoek member in server
                member = discord.utils.get(message.guild.members, name=member_name)
                if member:
                    await member.ban(reason=reason)
                    await message.channel.send(f"‚úÖ {member_name} is geband voor: {reason}")
                    if log_channel:
                        await log_channel.send(f"Ban log: {member_name} geband door {message.author} | Reason: {reason}")
                else:
                    await message.channel.send(f"‚ùå Kan {member_name} niet vinden.")

            except Exception as e:
                await message.channel.send(f"Fout bij ban: {e}")

        # ANNOUNCEMENT command
        elif "announcement" in content:
            try:
                announcement_text = content.split("announcement")[1].strip()
                await message.channel.send(f"üì¢ {announcement_text}")
                if log_channel:
                    await log_channel.send(f"Announcement log: '{announcement_text}' door {message.author}")
            except Exception as e:
                await message.channel.send(f"Fout bij announcement: {e}")

        else:
            await message.channel.send("‚ùå Command niet herkend. Gebruik ban ... reason ... of announcement ...")
